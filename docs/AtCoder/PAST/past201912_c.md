	```go 
    package main

    import (
        "bufio"
        "fmt"
        "os"
        "sort"
    )

    var r = bufio.NewReader(os.Stdin)

    func main() {
        solve()
    }

    func solve() {
        A := nis(6)
        sort.Ints(A)
        fmt.Println(A[3])

    }

    func nis(size int) []int {
        var n int
        res := make([]int, size)
        for i := range res {
            fmt.Fscan(r, &n)
            res[i] = n
        }
        return res
    }

    ```
